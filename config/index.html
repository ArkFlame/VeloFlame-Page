# VeloFlame PRO only (https://veloflame.com/)
# Analyzes nickname patterns to block bot attacks using similar names.
[nickname]
enabled = true
firewall = true
# If set to true, check will always be active, ignoring trigger-activations.
always-active = false
# Format: "hits:seconds"
# Logic: If hits reached in seconds, activate. Excess hits carry over to next period (capped at hits).
trigger-activations = [
    "30:60",
    "20:30",
    "10:15",
    "6:10",
    "3:5",
    "2:1"
]
# How many recent nicknames to compare against (Default: 10).
memory-size = 10
# Block if too many players have the same prefix. (eg zz_test, zz_bot, bot_test)
same-prefix-block = true
# How many same prefixes to trigger a block (Default: 3).
same-prefix-threshold = 3
# If true, players with similar nicknames will be blocked.
similarity-block = true
# It will only block after 3 nicknames have been flagged.
similarity-match-threshold = 3
# The percentage of similarity required to flag a name (0.70 = 70%).
similarity-threshold = 0.70
# If true, counts similarity only for names with the same length.
similarity-same-length = true
# Block when too many nicknames have the same length eg blah, bloh, bleh
same-length-block = true
# If there is more than 5 nicknames with the exact same length in memory, block
same-length-threshold = 5
logs = true
logs-trigger = false
whitelist = []
